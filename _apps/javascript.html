function draw_polygon(sides) {
    var i;
    for (i = 0; i < sides; i++) {
        var angle = 180 - (360/sides);
        mirobot.left(angle);
        mirobot.forward(100);
    }
}

mirobot.pendown();
draw_polygon(5);
mirobot.penup();
